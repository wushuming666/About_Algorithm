### T1

逆向思维，把所有可能的前缀存起来后再查找 `words` 中是否存在这些

```cpp
class Solution {
public:
    int countPrefixes(vector<string>& words, string s) {
        unordered_set<string> se;
        int n = s.size();
        for(int i = 0; i < 1; i ++)
            for(int j = 1; j <= n; j ++)
            {
                if(i + j > n) break;
                se.insert(s.substr(i, j));
            }
        int res = 0;
        for(auto &i : words) if(se.count(i)) res ++;
        return res;
    }
};

```

### T2

前缀和。特殊处理下右边为空（除数不能为0）

```cpp
typedef long long LL;
class Solution {
public:
    int minimumAverageDifference(vector<int>& nums) {
        int n = nums.size();
        if(!n) return 0;
        LL sum = 0;
        for(auto & i : nums) sum += i; //sum
        
        vector<LL> pre(n + 1);
        for(int i = 1; i <= n; i ++) pre[i] = pre[i-1] + nums[i-1];
        
        LL minn = 1e11; int idx = -1;
        for(int i = 1; i <= n; i ++){
            LL p1 = pre[i], p2 = sum - pre[i];
            if(i == n) p2 = 0;
            else p2 = p2 / (n - i);
            p1 /= i;
            LL tmp = abs(p1 - p2);
            if(minn > tmp)
            {
                minn = tmp;
                idx = i - 1;
            }
        }
        return idx;
    }
};

```

### T3

依题意模拟。每个警卫四个方向搜索。

```cpp
class Solution {
public:
    int countUnguarded(int n, int m, vector<vector<int>>& guards, vector<vector<int>>& walls) {
        vector<vector<int>> b;
        b.resize(n);
        for(auto &i : b) i = vector<int>(m);
        
        for(auto & i : walls){
            b[i[0]][i[1]] = -1;
        }
        for(auto & i : guards) b[i[0]][i[1]] = -1;
        
        for(auto & guard : guards){
            int x = guard[0], y = guard[1];
            for(int i = x - 1; i >= 0; --i) 
            {
                if(b[i][y] == -1) break;
                b[i][y] = 1;
            }
            for(int i = x + 1; i < n; i ++)
            {
                if(b[i][y] == -1) break;
                b[i][y] = 1;
            }
            for(int j = y + 1; j < m; j ++)
            {
                if(b[x][j] == -1) break;
                b[x][j] = 1;
            }
            for(int j = y - 1; j >= 0; j --)
            {
                if(b[x][j] == -1) break;
                b[x][j] = 1;
            }
        }
        
        int res = 0;
        for(int i = 0; i < n; i ++ ) for(int j = 0 ; j < m; j ++) if(!b[i][j]) res ++;
        return res;
        return 0;
    }
};

```

### T4

先把火蔓延的时间记录下来。

二分答案。

```cpp
typedef pair<int, int> PII;
class Solution {
public:
    int n, m;
    vector<vector<int>> cnt;
    int dx[4] = {0, 0, 1, -1}; int dy[4] = {1, -1, 0, 0};
    
    bool is_in(int x, int y){
        return x >= 0 && x < n && y >= 0 && y < m;
    }


    void bfs(vector<vector<int>>& grid)
    {
        for(auto &i : cnt) for(auto &j : i) j = 2e9;
        queue<PII> q;
        for(int i = 0; i < n; i ++ )
            for(int j = 0; j < m; j ++ )
                if(grid[i][j] == 1)
                {
                    q.push({i, j}); 
                    cnt[i][j] = 0;
                }

        //用cnt记录火蔓延的步数
        while(q.size())
        {
            auto [x, y] = q.front(); q.pop(); 
            for(int i = 0; i < 4; i ++)
            {
                int xx = x + dx[i], yy = y + dy[i];
                if(!is_in(xx, yy)) continue;
                if(grid[xx][yy] == 2) continue;
                if(cnt[xx][yy] != 2e9) continue; //已经visted
                cnt[xx][yy] = cnt[x][y] + 1;
                q.push({xx, yy});
            }
        }
    }

    //在起点等待u分钟后能否成功到达终点
    bool check(int u, vector<vector<int>>& grid){
        //将起点标为u后尝试bfs搜索到终点 某一点需要小于火到达这里的时间
        //终点除外  到达终点后可以直接逃离
        vector<vector<int>> peo(n, vector<int>(m));
        peo[0][0] = u;
        queue<PII> q;
        q.push({0, 0});
        while(q.size())
        {
            auto [x, y] = q.front(); q.pop();
            for(int i = 0; i < 4; i ++)
            {
                int xx = x + dx[i], yy = y + dy[i];
                if(!is_in(xx, yy)) continue;
                if(grid[xx][yy] == 2) continue;
                if(peo[xx][yy]) continue;
                
                peo[xx][yy] = peo[x][y] + 1;

                if(xx == n - 1 && yy == m - 1) //到达了安全屋
                {
                    if(peo[xx][yy] <= cnt[xx][yy]) return true;
                    return false;
                }
                if(peo[xx][yy] < cnt[xx][yy]){
                    
                    q.push({xx, yy});
                }
            }
        }
        return false;
    }

    int maximumMinutes(vector<vector<int>>& grid) {
        n = grid.size(), m = grid[0].size(); 
        cnt = vector<vector<int>>(n, vector<int> (m));
        bfs(grid); //存火到格子需要的时间

        //二分等待的时间

        if(check(1e8, grid)) return 1e9;
        int L = 0, R = 1e9;
        while(L < R){
            int mid = (L + R) >> 1;
            if(check(mid, grid)) L = mid + 1;
            else R = mid;
        }
        return L - 1;
    }
};
```
